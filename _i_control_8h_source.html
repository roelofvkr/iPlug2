867  IRECT DrawPressableTriangle(IGraphics&g, const IRECT& bounds, bool pressed, bool mouseOver, float angle)
868  {
869  float x1, x2, x3, y1, y2, y3;
870 
871  float theta = DegToRad(angle);
872 
873  IRECT handleBounds = GetAdjustedHandleBounds(bounds);
874 
875  // Center bounds around origin for rotation
876  float xT = handleBounds.L + handleBounds.W() * 0.5f;
877  float yT = handleBounds.T + handleBounds.H() * 0.5f;
878  IRECT centered = handleBounds.GetTranslated(-xT, -yT);
879 
880  // Do rotation and translate points back into view space
881  float c = cosf(theta);
882  float s = sinf(theta);
883  x1 = centered.L * c - centered.B * s + xT;
884  y1 = centered.L * s + centered.B * c + yT;
885  x2 = centered.MW() * c - centered.T * s + xT;
886  y2 = centered.MW() * s + centered.T * c + yT;
887  x3 = centered.R * c - centered.B * s + xT;
888  y3 = centered.R * s + centered.B * c + yT;
889 
890  if (pressed)
891  g.FillTriangle(GetColor(kPR), x1, y1, x2, y2, x3, y3);
892  else
893  {
