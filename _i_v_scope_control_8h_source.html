28 template <int MAXNC = 1, int MAXBUF = 128, int QUEUE_SIZE = 1024>
29 class IVScopeControl : public IControl

32 public:
33  static constexpr int kUpdateMessage = 0;

35  /* Data packet */
36  struct Data
37  {
38  int nchans = MAXNC;
39  float vals[MAXNC][MAXBUF] = {};
40 
41  bool AboveThreshold()
42  {
43  static const float threshold = (float) DBToAmp(-90.);
44 
45  float sum = 0.f;
46 
47  for(auto c = 0; c < MAXNC; c++)
48  {
49  for(auto s = 0; s < MAXBUF; s++)
50  {
51  sum += vals[c][s];
52  }
53  }
54 
55  return std::abs(sum) > threshold;
56  }
57  };

70  void Process(sample* inputs)

158  {

161  IRECT r = mWidgetBounds.GetPadded(-mPadding);
