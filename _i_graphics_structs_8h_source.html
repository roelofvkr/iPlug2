317  static IColor GetFromHSLA(float h, float s, float l, float a = 1.)
318  {
319  auto hue = [](float h, float m1, float m2)
320  {
321  if (h < 0) h += 1;
322  if (h > 1) h -= 1;
323  if (h < 1.0f / 6.0f)
324  return m1 + (m2 - m1) * h * 6.0f;
325  else if (h < 3.0f / 6.0f)
326  return m2;
327  else if (h < 4.0f / 6.0f)
328  return m1 + (m2 - m1) * (2.0f / 3.0f - h) * 6.0f;
329  return m1;
330  };
331 
332  IColor col;
333  h = std::fmodf(h, 1.0f);
334  if (h < 0.0f) h += 1.0f;
335  s = Clip(s, 0.0f, 1.0f);
336  l = Clip(l, 0.0f, 1.0f);
337  float m2 = l <= 0.5f ? (l * (1 + s)) : (l + s - l * s);
338  float m1 = 2 * l - m2;
339  col.R = static_cast<int>(Clip(hue(h + 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
340  col.G = static_cast<int>(Clip(hue(h, m1, m2), 0.0f, 1.0f) * 255.f);
341  col.B = static_cast<int>(Clip(hue(h - 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
342  col.A = static_cast<int>(a * 255.f);
343  return col;
344  }
